CREATE   PROCEDURE [geomon].[ServerRestartAlert] 
(
    @LookBackSeconds INT = 300
)
AS
BEGIN
    SET NOCOUNT ON;

    DECLARE @RestartTime DATETIME;

    SELECT @RestartTime = sqlserver_start_time
    FROM sys.dm_os_sys_info;

    -- Only proceed if restart is recent
    IF @RestartTime > DATEADD(SECOND, -@LookBackSeconds, GETDATE())
    BEGIN
        -- Insert only if not already logged
        IF NOT EXISTS (
            SELECT 1
            FROM [geomon].[ServerRestartHistory]
            WHERE restart_time = @RestartTime
        )
        BEGIN
            INSERT INTO [geomon].[ServerRestartHistory]
                (server_name, host_name, restart_time)
            VALUES
                (@@SERVERNAME, HOST_NAME(), @RestartTime);
        END;

        SELECT 
            @@SERVERNAME AS server_name,
            GETDATE() AS observation_time,
            'Restarted' AS agent_status,
            @RestartTime AS restart_time;
    END
END
GO

CREATE   PROCEDURE [dbo].[sp_CaptureRestartDetails]
AS
BEGIN
    SET NOCOUNT ON;

    DECLARE @LastStartTime DATETIME, @PrevShutdown DATETIME, @LogJSON NVARCHAR(MAX);

    -- 1. Get current SQL Start Time [cite: 17]
    SELECT @LastStartTime = sqlserver_start_time FROM sys.dm_os_sys_info;

    -- 2. Find the last recorded 'shutdown' message in the ErrorLog [cite: 3]
    DECLARE @LogTable TABLE (LogDate DATETIME, ProcessInfo VARCHAR(50), Text NVARCHAR(MAX));
    INSERT INTO @LogTable EXEC xp_readerrorlog 0, 1, N'SQL Server shutdown has been initiated';
    SELECT TOP 1 @PrevShutdown = LogDate FROM @LogTable ORDER BY LogDate DESC;

    -- 3. Gather 50 log entries for AI Evidence [cite: 114]
    DECLARE @Evidence TABLE (LogDate DATETIME, ProcInfo VARCHAR(50), Text NVARCHAR(MAX));
    INSERT INTO @Evidence EXEC xp_readerrorlog 0; 
    
    -- Format as JSON so Claude API can parse it easily [cite: 114]
    SET @LogJSON = (SELECT TOP 50 Text FROM @Evidence ORDER BY LogDate ASC FOR JSON PATH);

    -- 4. Insert data using minutes for uptime [cite: 10, 158]
    IF NOT EXISTS (SELECT 1 FROM dbo.SQLServerRestartHistory WHERE SQLStartTime = @LastStartTime)
    BEGIN
        INSERT INTO dbo.SQLServerRestartHistory 
            (ServerName, SQLStartTime, PreviousShutdownTime, Uptime_Minutes, Downtime_Minutes, EvidenceSummary, RestartType, ProcessingStatus)
        VALUES (
            @@SERVERNAME, 
            @LastStartTime, 
            @PrevShutdown, 
            DATEDIFF(MINUTE, @PrevShutdown, @LastStartTime), -- Uptime in Minutes [cite: 158]
            DATEDIFF(MINUTE, @PrevShutdown, @LastStartTime), -- Downtime in Minutes [cite: 10]
            @LogJSON, 
            'PENDING_AI_ANALYSIS', 
            'PENDING_PUSH'
        );
    END
END;
GO
